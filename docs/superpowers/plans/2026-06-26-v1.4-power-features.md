```js
{
  id: string,           // timestamp-based unique id
  timestamp: number,    // Unix ms
  url: string,          // page URL
  pageTitle: string,    // document.title
  framework: string,
  language: string,
  style: string,
  className: string,
  elementCount: number,
  elements: object      // { inputs, buttons, links, selects, textareas }
}
```

```js
const MAX_SNAPSHOTS = 20;

function createSnapshot({ url, pageTitle, framework, language, style, className, elements }) {
  const allElements = ['inputs', 'buttons', 'links', 'selects', 'textareas']
    .flatMap(g => elements[g] || []);
  return {
    id: Date.now() + '-' + Math.random().toString(36).slice(2, 7),
    timestamp: Date.now(),
    url,
    pageTitle,
    framework,
    language,
    style,
    className,
    elementCount: allElements.length,
    elements,
  };
}

function loadSnapshots(storage, cb) {
  storage.get('pomHistory', ({ pomHistory }) => {
    cb(pomHistory || []);
  });
}

function addSnapshot(snapshot, storage, cb) {
  loadSnapshots(storage, (existing) => {
    const updated = [...existing, snapshot];
    const capped = updated.length > MAX_SNAPSHOTS ? updated.slice(updated.length - MAX_SNAPSHOTS) : updated;
    storage.set({ pomHistory: capped }, cb);
  });
}

function removeSnapshot(id, storage, cb) {
  loadSnapshots(storage, (existing) => {
    storage.set({ pomHistory: existing.filter(s => s.id !== id) }, cb);
  });
}

function exportSnapshots(storage, cb) {
  loadSnapshots(storage, (snaps) => {
    cb(JSON.stringify(snaps, null, 2));
  });
}

function importSnapshots(jsonString, storage, cb) {
  let incoming;
  try { incoming = JSON.parse(jsonString); } catch (e) { if (cb) cb(); return; }
  if (!Array.isArray(incoming)) { if (cb) cb(); return; }

  loadSnapshots(storage, (existing) => {
    const existingIds = new Set(existing.map(s => s.id));
    const newOnes = incoming.filter(s => s.id && !existingIds.has(s.id));
    const merged = [...existing, ...newOnes];
    const capped = merged.length > MAX_SNAPSHOTS ? merged.slice(merged.length - MAX_SNAPSHOTS) : merged;
    storage.set({ pomHistory: capped }, cb);
  });
}

module.exports = { createSnapshot, addSnapshot, loadSnapshots, removeSnapshot, exportSnapshots, importSnapshots };
```

```js
// ---- Inlined history helpers (mirrors src/history.js) ----
const POM_HISTORY_MAX = 20;

function historyCreateSnapshot(url, pageTitle, framework, language, style, className, elements) {
  const allEls = ['inputs','buttons','links','selects','textareas'].flatMap(g => elements[g] || []);
  return {
    id: Date.now() + '-' + Math.random().toString(36).slice(2, 7),
    timestamp: Date.now(),
    url, pageTitle, framework, language, style, className,
    elementCount: allEls.length,
    elements,
  };
}

function historySave(snapshot, cb) {
  chrome.storage.local.get('pomHistory', ({ pomHistory }) => {
    const existing = pomHistory || [];
    const updated = [...existing, snapshot];
    const capped = updated.length > POM_HISTORY_MAX ? updated.slice(updated.length - POM_HISTORY_MAX) : updated;
    chrome.storage.local.set({ pomHistory: capped }, cb);
  });
}

function historyLoad(cb) {
  chrome.storage.local.get('pomHistory', ({ pomHistory }) => cb(pomHistory || []));
}

function historyRemove(id, cb) {
  historyLoad((existing) => {
    chrome.storage.local.set({ pomHistory: existing.filter(s => s.id !== id) }, cb);
  });
}
```

- [ ] **Step 4: Add `renderHistory` function**

```js
function renderHistory() {
  historyLoad((snaps) => {
    const list = document.getElementById('history-list');
    const empty = document.getElementById('history-empty');
    list.innerHTML = '';

    if (snaps.length === 0) {
      empty.style.display = 'block';
      list.appendChild(empty);
      return;
    }

    // Group by URL
    const groups = {};
    for (const snap of [...snaps].reverse()) {
      const key = snap.url || 'Unknown page';
      if (!groups[key]) groups[key] = [];
      groups[key].push(snap);
    }

    for (const [url, groupSnaps] of Object.entries(groups)) {
      const groupLabel = document.createElement('div');
      groupLabel.className = 'history-group-label';
      groupLabel.textContent = url.replace(/^https?:\/\//, '').slice(0, 50);
      list.appendChild(groupLabel);

      for (const snap of groupSnaps) {
        const item = document.createElement('div');
        item.className = 'history-item';

        const title = document.createElement('div');
        title.className = 'history-item-title';
        title.textContent = snap.className;

        const meta = document.createElement('div');
        meta.className = 'history-item-meta';
        const date = new Date(snap.timestamp).toLocaleString();
        meta.innerHTML = `<span>${date}</span><span>${snap.framework} / ${snap.language}</span><span>${snap.elementCount} elements</span>`;

        const actions = document.createElement('div');
        actions.className = 'history-item-actions';

        const restoreBtn = document.createElement('button');
        restoreBtn.textContent = 'Restore';
        restoreBtn.addEventListener('click', (e) => {
          e.stopPropagation();
          restoreSnapshot(snap);
        });

        const deleteBtn = document.createElement('button');
        deleteBtn.className = 'btn-delete';
        deleteBtn.textContent = 'Delete';
        deleteBtn.addEventListener('click', (e) => {
          e.stopPropagation();
          historyRemove(snap.id, () => renderHistory());
        });

        actions.appendChild(restoreBtn);
        actions.appendChild(deleteBtn);
        item.appendChild(title);
        item.appendChild(meta);
        item.appendChild(actions);
        list.appendChild(item);
      }
    }
  });
}
```

function restoreSnapshot(snap) {
  currentElements = snap.elements;
  currentFramework = snap.framework;

document.getElementById('import-snapshot-input').addEventListener('change', (e) => {
  const file = e.target.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = (evt) => {
    const json = evt.target.result;
    let incoming;
    try { incoming = JSON.parse(json); } catch { return; }
    if (!Array.isArray(incoming)) return;

    historyLoad((existing) => {
      const existingIds = new Set(existing.map(s => s.id));
      const newOnes = incoming.filter(s => s.id && !existingIds.has(s.id));
      const merged = [...existing, ...newOnes];
      const capped = merged.length > POM_HISTORY_MAX ? merged.slice(merged.length - POM_HISTORY_MAX) : merged;
      chrome.storage.local.set({ pomHistory: capped }, () => renderHistory());
    });
  };
  reader.readAsText(file);
  e.target.value = ''; // reset so same file can be re-imported
});
