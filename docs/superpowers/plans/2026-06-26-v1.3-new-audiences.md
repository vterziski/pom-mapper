### What this adds
A `csharp` language branch in `toLocatorString` for both `playwright` and `selenium` frameworks.

**Playwright C# locator patterns:**
- `testid` → `_page.GetByTestId("value")`
- `label`  → `_page.GetByLabel("value")`
- `title`  → `_page.GetByTitle("value")`
- `id`     → `_page.Locator("#value")`
- `name`   → `_page.Locator("[name=\"value\"]")`
- `role`   → `_page.GetByRole(AriaRole.Button, new() { Name = "value" })`

**Selenium C# locator patterns:**
- `id`     → `_driver.FindElement(By.Id("value"))`
- `name`   → `_driver.FindElement(By.Name("value"))`
- `testid` → `_driver.FindElement(By.CssSelector("[data-testid=\"value\"]"))`
- `label`  → `_driver.FindElement(By.CssSelector("[aria-label=\"value\"]"))`
- `title`  → `_driver.FindElement(By.CssSelector("[title=\"value\"]"))`
- `role`   → `_driver.FindElement(By.CssSelector("[role=\"value\"]"))`

```js
const { toLocatorString } = require('../locators');

const GROUP_LABELS = { inputs: 'Inputs', buttons: 'Buttons', links: 'Links', selects: 'Selects', textareas: 'Textareas' };
const GROUPS = ['inputs', 'buttons', 'links', 'selects', 'textareas'];

function generate(elements, className) {
  const activeGroups = GROUPS.filter(g => elements[g] && elements[g].length > 0);
  const lines = [
    'using Microsoft.Playwright;',
    '',
    `public class ${className}`,
    '{',
    '    private readonly IPage _page;',
    '',
    `    public ${className}(IPage page) => _page = page;`,
  ];

  for (const g of activeGroups) {
    lines.push('');
    lines.push(`    // ${GROUP_LABELS[g]}`);
    for (const el of elements[g]) {
      const loc = toLocatorString(el.locatorData, 'playwright', 'csharp');
      const propName = el.name.charAt(0).toUpperCase() + el.name.slice(1);
      lines.push(`    public ILocator ${propName} => ${loc};`);
    }
  }

  lines.push('}');
  return lines.join('\n');
}

module.exports = { generate };
```

```js
const GROUP_LABELS = { inputs: 'Inputs', buttons: 'Buttons', links: 'Links', selects: 'Selects', textareas: 'Textareas' };
const GROUPS = ['inputs', 'buttons', 'links', 'selects', 'textareas'];

function wdioSelector(locatorData) {
  const { strategy, value, role } = locatorData;
  if (strategy === 'testid') return `[data-testid="${value}"]`;
  if (strategy === 'label')  return `[aria-label="${value}"]`;
  if (strategy === 'id')     return `#${value}`;
  if (strategy === 'name')   return `[name="${value}"]`;
  if (strategy === 'title')  return `[title="${value}"]`;
  if (strategy === 'role')   return `[role="${role}"]`;
  return value;
}

function generate(elements, className) {
  const activeGroups = GROUPS.filter(g => elements[g] && elements[g].length > 0);
  const lines = [`class ${className} {`];

  for (const g of activeGroups) {
    lines.push('');
    lines.push(`  // ${GROUP_LABELS[g]}`);
    for (const el of elements[g]) {
      const sel = wdioSelector(el.locatorData);
      lines.push(`  get ${el.name}() { return $('${sel}'); }`);
    }
  }

  lines.push('}');
  lines.push('');
  lines.push(`module.exports = new ${className}();`);
  return lines.join('\n');
}

module.exports = { generate };
```

```js
const GROUPS = ['inputs', 'buttons', 'links', 'selects', 'textareas'];

function robotSelector(locatorData) {
  const { strategy, value, role } = locatorData;
  if (strategy === 'testid') return `[data-testid="${value}"]`;
  if (strategy === 'label')  return `[aria-label="${value}"]`;
  if (strategy === 'id')     return `id=${value}`;
  if (strategy === 'name')   return `[name="${value}"]`;
  if (strategy === 'title')  return `[title="${value}"]`;
  if (strategy === 'role')   return `[role="${role}"]`;
  return value;
}

function toScreamingSnake(name) {
  return name
    .replace(/([A-Z])/g, '_$1')
    .replace(/^_/, '')
    .toUpperCase();
}

function toTitleCase(name) {
  return name
    .replace(/([A-Z])/g, ' $1')
    .replace(/^./, s => s.toUpperCase())
    .trim();
}

function generate(elements, className) {
  const allElements = GROUPS.flatMap(g => elements[g] || []);
  const lines = [
    '*** Settings ***',
    'Library    Browser',
    '',
    '*** Variables ***',
  ];

  for (const el of allElements) {
    const varName = '${' + toScreamingSnake(el.name) + '}';
    const sel = robotSelector(el.locatorData);
    lines.push(`${varName.padEnd(30)}${sel}`);
  }

  lines.push('');
  lines.push('*** Keywords ***');

  for (const el of elements.inputs || []) {
    const varName = '${' + toScreamingSnake(el.name) + '}';
    const kwName = 'Fill ' + toTitleCase(el.name);
    lines.push('');
    lines.push(kwName);
    lines.push('    [Arguments]    ${value}');
    lines.push(`    Fill Text    ${varName}    \${value}`);
  }

  for (const el of elements.buttons || []) {
    const varName = '${' + toScreamingSnake(el.name) + '}';
    const kwName = 'Click ' + toTitleCase(el.name);
    lines.push('');
    lines.push(kwName);
    lines.push(`    Click    ${varName}`);
  }

  for (const el of elements.links || []) {
    const varName = '${' + toScreamingSnake(el.name) + '}';
    const kwName = 'Click ' + toTitleCase(el.name);
    lines.push('');
    lines.push(kwName);
    lines.push(`    Click    ${varName}`);
  }

  return lines.join('\n');
}

module.exports = { generate };
```

```js
const GROUPS = ['inputs', 'buttons', 'links', 'selects', 'textareas'];

function robotSelector(locatorData) {
  const { strategy, value, role } = locatorData;
  if (strategy === 'testid') return `[data-testid="${value}"]`;
  if (strategy === 'label')  return `[aria-label="${value}"]`;
  if (strategy === 'id')     return `id=${value}`;
  if (strategy === 'name')   return `[name="${value}"]`;
  if (strategy === 'title')  return `[title="${value}"]`;
  if (strategy === 'role')   return `[role="${role}"]`;
  return value;
}

function toSnakeCase(name) {
  return name.replace(/([A-Z])/g, '_$1').replace(/^_/, '').toLowerCase();
}

function generate(elements, className) {
  const lines = [
    'from robot.api.deco import keyword',
    'from Browser import Browser',
    '',
    `class ${className}:`,
    '    def __init__(self):',
    '        self.browser = Browser()',
    '',
  ];

  const allElements = GROUPS.flatMap(g => elements[g] || []);
  for (const el of allElements) {
    const methodName = toSnakeCase(el.name);
    const sel = robotSelector(el.locatorData);
    lines.push('    @keyword');
    lines.push(`    def ${methodName}(self):`);
    lines.push(`        return "${sel}"`);
    lines.push('');
  }

  return lines.join('\n');
}

module.exports = { generate };
```

```js
function onFrameworkChange(framework) {
  currentFramework = framework;

  // Show/hide style toggle
  const styleRow = document.getElementById('style-toggle-row');
  if (framework === 'wdio' || framework === 'robot') {
    styleRow.classList.remove('hidden');
    // Update style toggle labels
    const styleBtns = document.querySelectorAll('#style-selector .seg-btn');
    if (framework === 'robot') {
      styleBtns[0].textContent = 'Resource File';
      styleBtns[0].dataset.value = 'resource';
      styleBtns[1].textContent = 'Python Library';
      styleBtns[1].dataset.value = 'python';
      currentStyle = 'resource';
    } else {
      styleBtns[0].textContent = 'POM Style';
      styleBtns[0].dataset.value = 'pom';
      styleBtns[1].textContent = 'Native Style';
      styleBtns[1].dataset.value = 'native';
      currentStyle = 'pom';
    }
    // Reset active
    styleBtns.forEach(b => b.classList.remove('active'));
    styleBtns[0].classList.add('active');
  } else {
    styleRow.classList.add('hidden');
  }

  // Disable C# for Cypress, WDIO, Robot
  const csharpBtn = document.querySelector('#language-selector [data-value="csharp"]');
  const blockedForCsharp = ['cypress', 'wdio', 'robot'];
  csharpBtn.disabled = blockedForCsharp.includes(framework);
  if (csharpBtn.disabled && currentLanguage === 'csharp') {
    // Reset to TypeScript
    currentLanguage = 'ts';
    document.querySelectorAll('#language-selector .seg-btn').forEach(b => {
      b.classList.toggle('active', b.dataset.value === 'ts');
    });
  }

  // Disable Java and Python for WDIO (JS only)

        b.classList.toggle('active', lang === 'js');
```

```markdown
# POM Mapper — User Guide

## Quick Start

1. Install POM Mapper from the [Chrome Web Store](https://chromewebstore.google.com/detail/pom-mapper/hdjcdjdfeejjkhlnkpnimicoiffpchbp) or [Firefox Add-ons](https://addons.mozilla.org/en-US/firefox/addon/pom-mapper/)
2. Navigate to any web page you want to map
3. Click the POM Mapper icon in your toolbar
4. Enter a class name (e.g. `LoginPage`)
5. Select your framework and language
6. Click **Map Page**
7. Click **Export** to download the generated file

---

## Accumulative Scanning

POM Mapper accumulates elements across multiple scans. Open a modal or dropdown, click **Map Page** again, and the new elements are added to the existing results.

To start fresh, click **Scan Again**.

---

## Frameworks & Languages

### Playwright
Supports TypeScript, JavaScript, Java, Python, and C#.

Uses Playwright's preferred locator methods: `getByTestId`, `getByLabel`, `getByRole`, `getByTitle`, `locator`.

### Selenium
Supports TypeScript, JavaScript, Java, Python, and C#.

Uses `By.id`, `By.name`, `By.css`, `By.cssSelector` depending on language.

### Cypress
Supports TypeScript and JavaScript.

Uses `cy.get()` with CSS selectors and attribute matchers.

### WebdriverIO
Supports JavaScript only.

**Export styles:**
- **POM Style** — class with `get` accessors using `$()`, exported as `module.exports = new LoginPage()`
- **Native Style** — class extending `Page`, using ES module exports as per WDIO official docs

### Robot Framework
**Export styles:**
- **Resource File** — `.robot` file with `*** Variables ***` and `*** Keywords ***` sections using the Browser library
- **Python Library** — Python class with `@keyword` decorated methods, importable as a Robot Framework library

### Language availability

| Language | Playwright | Selenium | Cypress | WebdriverIO | Robot |
|----------|-----------|---------|---------|------------|-------|
| TypeScript | ✅ | ✅ | ✅ | — | — |
| JavaScript | ✅ | ✅ | ✅ | ✅ | — |
| Java | ✅ | ✅ | — | — | — |
| Python | ✅ | ✅ | — | — | ✅ |
| C# | ✅ | ✅ | — | — | — |

## Element Picker

Click **Pick Elements** to enter pick mode. Hover over elements on the page — a blue highlight shows what will be selected. Click an element to add it to your results. Click **Stop Picking** when done.

Picked elements appear in the results list alongside scanned elements and are included in the export.

---

## Rename & Exclude Elements

**Rename:** Hover over any element row and click the ✎ pencil icon. Type a new name and press Enter. The export uses your name. Renamed rows show a small blue dot.

**Exclude:** Uncheck the checkbox on any element row to remove it from the export. Use **Select all / Deselect all** to toggle all elements at once.

The **Export** button shows the current count of selected elements.

---

## iframe Scanning

If POM Mapper detects iframes on the page, a yellow **Iframes detected** panel appears above the results. Click to expand it, check the iframes you want to scan, and click **Scan Selected Iframes**. Elements from those iframes are merged into your results.

Cross-origin iframes (e.g. embedded third-party widgets) cannot be scanned and are shown as disabled.

---

## Locator Quality Score

Each element row shows a mini score bar indicating locator stability:

- 🟢 **Strong** (full bar) — `data-testid`, `aria-label`, `id`
- 🟡 **Fair** (60%) — `name`, `title`, `role + text`
- 🔴 **Weak** (25%) — positional or class-based selectors

Click a Fair or Weak bar to see 2–3 alternative locators ranked by stability. Select one to replace the original.

---

## Shadow DOM & Salesforce

On Salesforce Lightning pages, POM Mapper automatically detects the environment and enables **Deep Scan** mode, which pierces shadow DOM boundaries.

You can also enable Deep Scan manually using the toggle that appears after clicking the Salesforce badge.

## AI Self-Healing

When locators break after a UI change:

1. Go to **Settings** (⚙️ icon) and enter your API key
2. Select your provider: **OpenAI** (`sk-...`) or **Anthropic** (`sk-ant-...`)
3. From the results view, click **✨ Heal Locators**
4. The extension rescans the page, identifies broken locators, and suggests replacements
5. Approve each suggestion individually

API keys are stored locally in browser storage and never transmitted to anyone other than your chosen AI provider.
```
