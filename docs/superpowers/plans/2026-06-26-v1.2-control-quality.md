## Task 1: Quality Scorer Module

**Files:**
- Create: `src/quality.js`
- Create: `tests/quality.test.js`

### What this module does
Scores a locator based on its strategy and returns a tier (`strong`, `fair`, `weak`), a numeric score (0–100), and a list of alternative locator data objects ranked by stability.

- **strong** (score 90–100): `testid`, `label`, `id`
- **fair** (score 50–89): `name`, `title`, `role`
- **weak** (score 0–49): anything else (positional, class-based)

```js
const STRATEGY_SCORES = {
  testid: 95,
  label:  90,
  id:     88,
  name:   70,
  title:  65,
  role:   55,
};

const ROLE_MAP = {
  button: 'button', link: 'link', select: 'combobox',
  input: 'textbox', textarea: 'textbox',
};

function scoreLocator(locatorData) {
  const score = STRATEGY_SCORES[locatorData.strategy] ?? 20;
  const tier = score >= 90 ? 'strong' : score >= 50 ? 'fair' : 'weak';
  return { tier, score, alternatives: [] };
}

function getAllCandidates(el, type) {
  const candidates = [];

  const testId = el.getAttribute('data-testid');
  if (testId) candidates.push({ strategy: 'testid', value: testId, rawName: testId });

  const ariaLabel = el.getAttribute('aria-label');
  if (ariaLabel) candidates.push({ strategy: 'label', value: ariaLabel, rawName: ariaLabel });

  const id = el.getAttribute('id');
  if (id) candidates.push({ strategy: 'id', value: id, rawName: id });

  const name = el.getAttribute('name');
  if (name) candidates.push({ strategy: 'name', value: name, rawName: name });

  const title = el.getAttribute('title');
  if (title) candidates.push({ strategy: 'title', value: title, rawName: title });

  const text = (el.textContent || '').trim();
  if (text) candidates.push({ strategy: 'role', role: ROLE_MAP[type] || type, value: text, rawName: text });

  return candidates;
}

function scoreLocatorWithAlternatives(locatorData, el, type) {
  const base = scoreLocator(locatorData);
  if (!el) return base;

  const all = getAllCandidates(el, type);
  const alternatives = all
    .filter(c => c.strategy !== locatorData.strategy)
    .map(c => ({ ...c, ...scoreLocator(c) }))
    .sort((a, b) => b.score - a.score)
    .slice(0, 3);

  return { ...base, alternatives };
}

module.exports = { scoreLocator, getAllCandidates, scoreLocatorWithAlternatives };
```

```js
function detectIframes(doc) {
  const iframes = Array.from(doc.querySelectorAll('iframe'));
  return iframes.map(iframe => {
    const src = iframe.getAttribute('src') || '';
    const title = iframe.getAttribute('title') || '';
    let sameOrigin = false;
    try {
      sameOrigin = !!(iframe.contentDocument && iframe.contentDocument.body);
    } catch (e) {
      sameOrigin = false;
    }
    return { src, title, label: title || src, sameOrigin };
  });
}
```

```js
const BLOCKED_TAGS = new Set(['HTML', 'BODY']);

function isPickable(tagName, isOverlay = false) {
  if (isOverlay) return false;
  if (BLOCKED_TAGS.has(tagName)) return false;
  return true;
}

function getPickerLabel(el) {
  const testId = el.getAttribute('data-testid');
  if (testId) return `[data-testid="${testId}"]`;

  const label = el.getAttribute('aria-label');
  if (label) return `[aria-label="${label}"]`;

  const id = el.getAttribute('id');
  if (id) return `#${id}`;

  const name = el.getAttribute('name');
  if (name) return `[name="${name}"]`;

  const text = (el.textContent || '').trim();
  if (text) return text.slice(0, 40);

  return el.tagName;
}

function getHighlightStyle(rect, scrollY = 0) {
  return [
    `position: absolute`,
    `top: ${rect.top + scrollY}px`,
    `left: ${rect.left}px`,
    `width: ${rect.width}px`,
    `height: ${rect.height}px`,
    `pointer-events: none`,
    `border: 2px solid #4a90e2`,
    `background: rgba(74,144,226,0.08)`,
    `border-radius: 3px`,
    `z-index: 2147483647`,
    `box-sizing: border-box`,
  ].join('; ');
}

module.exports = { isPickable, getPickerLabel, getHighlightStyle };
```

```js
  document.addEventListener('click', _onPickerClick, true);

function deactivatePickerOverlay() {
  if (!_pickerActive) return;
  _pickerActive = false;
  document.removeEventListener('mouseover', _onPickerHover, true);
  document.removeEventListener('click', _onPickerClick, true);
  if (_pickerOverlay) { _pickerOverlay.remove(); _pickerOverlay = null; }
  if (_pickerTooltip) { _pickerTooltip.remove(); _pickerTooltip = null; }

function _getPickerLabel(el) {
  var testId = el.getAttribute && el.getAttribute('data-testid');
  if (testId) return '[data-testid="' + testId + '"]';
  var label = el.getAttribute && el.getAttribute('aria-label');
  if (label) return '[aria-label="' + label + '"]';
  var id = el.getAttribute && el.getAttribute('id');
  if (id) return '#' + id;
  var name = el.getAttribute && el.getAttribute('name');
  if (name) return '[name="' + name + '"]';
  var text = (el.textContent || '').trim().slice(0, 40);
  if (text) return text;
  return el.tagName;

var _lastHovered = null;
function _onPickerHover(e) {
  var el = e.target;
  if (!el || el.id === '__pom_picker_overlay__' || el.id === '__pom_picker_tooltip__') return;
  if (el.tagName === 'HTML' || el.tagName === 'BODY') return;

  _lastHovered = el;
  var rect = el.getBoundingClientRect();

  _pickerOverlay.style.cssText = [

    'border-radius: 3px', 'z-index: 2147483647', 'box-sizing: border-box'

  _pickerTooltip.textContent = _getPickerLabel(el);
  _pickerTooltip.style.display = 'block';
  _pickerTooltip.style.top = (rect.bottom + 4) + 'px';
  _pickerTooltip.style.left = rect.left + 'px';
}

function _onPickerClick(e) {
  e.preventDefault();
  e.stopPropagation();
  var el = _lastHovered;
  if (!el) return;

  var label = _getPickerLabel(el);
  chrome.runtime.sendMessage({ action: 'pickerElementSelected', label: label, tagName: el.tagName });
}
```

```js
function toggleSuggestions(el, row, qualityWrap) {
  const existing = row.querySelector('.quality-suggestions');
  if (existing) { existing.remove(); return; }

  const alternatives = getAlternativeLocators(el.locatorData);
  if (alternatives.length === 0) return;

  const panel = document.createElement('div');
  panel.className = 'quality-suggestions';

  const title = document.createElement('div');
  title.style.cssText = 'font-weight:600;margin-bottom:4px;color:#555';
  title.textContent = 'Alternative locators:';
  panel.appendChild(title);

  for (const alt of alternatives) {
    const opt = document.createElement('div');
    opt.className = 'suggestion-option';

    const radio = document.createElement('input');
    radio.type = 'radio';
    radio.name = 'suggestion-' + el.name;

    const locStr = document.createElement('span');
    locStr.className = 'suggestion-locator';
    locStr.textContent = strategyToLocatorPreview(alt);

    const tierBadge = document.createElement('span');
    const { tier: altTier } = scoreLocatorInline(alt);
    tierBadge.style.cssText = 'font-size:10px;margin-left:auto;color:' +
      (altTier === 'strong' ? '#4caf50' : altTier === 'fair' ? '#f57c00' : '#e53935');
    tierBadge.textContent = altTier;

    opt.appendChild(radio);
    opt.appendChild(locStr);
    opt.appendChild(tierBadge);

    radio.addEventListener('change', () => {
      el.locatorData = alt;
      const fresh = renderElementRow(el);
      row.replaceWith(fresh);
    });

    panel.appendChild(opt);
  }

  // Insert after rowMain inside the row
  row.querySelector('.row-main').after(panel);
}

function getAlternativeLocators(locatorData) {
  // Returns up to 3 alternative strategies based on what's available in the locatorData
  // In a real scan, the element has only one locatorData — alternatives come from re-scoring
  // For the popup we show the strategies we know about but didn't pick
  const all = [
    { strategy: 'testid', value: locatorData._testid },
    { strategy: 'label',  value: locatorData._label },
    { strategy: 'id',     value: locatorData._id },
    { strategy: 'name',   value: locatorData._name },
    { strategy: 'title',  value: locatorData._title },
  ].filter(a => a.value && a.strategy !== locatorData.strategy);

  return all
    .map(a => ({ ...a, rawName: a.value }))
    .sort((a, b) => (STRATEGY_SCORES[b.strategy] || 0) - (STRATEGY_SCORES[a.strategy] || 0))
    .slice(0, 3);
}

function strategyToLocatorPreview(locatorData) {
  const { strategy, value, role } = locatorData;
  if (strategy === 'testid') return `[data-testid="${value}"]`;
  if (strategy === 'label')  return `[aria-label="${value}"]`;
  if (strategy === 'id')     return `#${value}`;
  if (strategy === 'name')   return `[name="${value}"]`;
  if (strategy === 'title')  return `[title="${value}"]`;
  if (strategy === 'role')   return `[role="${role}"]`;
  return value;
}
```
